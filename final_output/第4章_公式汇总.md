## 4.1 诊断问题的物理约束与降维动机

$$\tilde{\varepsilon}_r(\omega) = 1 - \frac{\omega_p^2}{\omega(\omega + j\nu_e)} = \left( 1 - \frac{\omega_p^2}{\omega^2 + \nu_e^2} \right) + j \left( \frac{\nu_e}{\omega} \frac{\omega_p^2}{\omega^2 + \nu_e^2} \right) \tag{4-1}$$

$$\left( \frac{\nu_e}{\omega} \right)^2 \ll 1 \tag{4-2}$$

$$\frac{1}{\omega^2 + \nu_e^2} \approx \frac{1}{\omega^2} \left[ 1 - \left(\frac{\nu_e}{\omega}\right)^2 + O\left(\left(\frac{\nu_e}{\omega}\right)^4\right) \right] \tag{4-3}$$

$$\varepsilon'(\omega) \approx \underbrace{\left( 1 - \frac{\omega_p^2}{\omega^2} \right)}_{\text{主导项(无碰撞)}} + \underbrace{\frac{\omega_p^2}{\omega^2} \left( \frac{\nu_e}{\omega} \right)^2}_{\text{碰撞修正项}} \tag{4-4}$$

$$\tau_g(\omega) = \frac{d}{d\omega} \left[ \frac{\omega d}{c} \text{Re}\left( \sqrt{\tilde{\varepsilon}_r(\omega)} \right) \right] \tag{4-5}$$

$$J(n_e, \nu_e) = \sum_{i=1}^{N} w_i \left\| \tau_{meas}(f_i) - \tau_{theory}(f_i; n_e, \nu_e) \right\|^2 \tag{4-6}$$

$$\mathbf{J}_{k} = \left[ \frac{\partial r_k}{\partial n_e}, \quad \frac{\partial r_k}{\partial \nu_e} \right] \tag{4-7}$$

$$\frac{\partial \tau_{theory}}{\partial n_e} \sim O\left( \frac{d}{c} \cdot \frac{1}{\omega^2 \sqrt{1-(f_p/f)^2}} \right) \approx 10^{-21} \text{ s·m}^3 \tag{4-8}$$

$$\frac{\partial \tau_{theory}}{\partial \nu_e} \sim O\left( \frac{\omega_p^2}{\omega^4} \cdot \frac{\nu_e}{\omega} \cdot \frac{d}{c} \right) \approx 10^{-32} \text{ s}^2 \tag{4-9}$$

$$\text{cond}(\mathbf{H}) \sim \frac{\sum (\partial \tau / \partial n_e)^2}{\sum (\partial \tau / \partial \nu_e)^2} \sim 10^{22} \tag{4-10}$$

$$\Delta J = J(n_e^*, 10\text{ GHz}) - J(n_e^*, 0.1\text{ GHz}) < N \sigma^2 \tag{4-11}$$

$$\delta_{\nu} = \frac{\nu_e^{preset} - \nu_e^{true}}{\nu_e^{true}} \tag{4-12}$$

$$\Delta \tau_{sys} \approx \frac{\partial \tau}{\partial \nu_e} \cdot (\nu_e^{preset} - \nu_e^{true}) \tag{4-13}$$

$$|\Delta \tau_{sys}| \sim 10^{-32} \times (0.5 \times 1.5 \times 10^9) \approx 0.01 \text{ ps} \tag{4-14}$$

$$\Delta n_e \sim \frac{\Delta \tau_{sys}}{\partial \tau / \partial n_e} \approx 1\% \times n_e \tag{4-15}$$

## 4.2 强色散信号的高分辨率特征提取

$$\delta f = |\alpha| T_w \le \beta \cdot \Delta f_{target} \tag{4-16}$$

$$T_w \le \frac{\beta \cdot \Delta f_{target}}{|\alpha|} = \frac{0.1 \times 10^4}{10^6} = 1 \text{ ms} \tag{4-17}$$

$$N_w = \left\lfloor \frac{T_m - T_w}{T_{step}} \right\rfloor + 1 \approx 32 \tag{4-18}$$

$$f_{probe}(t_i) = f_0 + K \cdot t_i \tag{4-19}$$

$$f_{beat,i} = K \cdot \tau_g(t_i) \tag{4-20}$$

$$\tau_{meas,i} = \frac{f_{beat,i}}{K} \tag{4-21}$$

$$A_i = \sqrt{\frac{1}{N_{sam}}\sum_{n=1}^{N_{sam}}|s_D(t_{i,n})|^2} \tag{4-22}$$

$$s_D(t) = \sum_{k=0}^{K-1} A_k \exp\{j[2\pi f_D^{(k)} t + \phi_k]\} + n(t) \tag{4-23}$$

$$\mathbf{X} = \begin{bmatrix}
s_D(0) & s_D(1) & \cdots & s_D(M-1) \\
s_D(1) & s_D(2) & \cdots & s_D(M) \\
\vdots & \vdots & \ddots & \vdots \\
s_D(L-1) & s_D(L) & \cdots & s_D(N_w-1)
\end{bmatrix} \tag{4-24}$$

$$\mathbf{R}_f = \frac{1}{M}\mathbf{X}\mathbf{X}^H \tag{4-25}$$

$$\mathbf{R}_x = \frac{1}{2}(\mathbf{R}_f + \mathbf{J}\mathbf{R}_f^*\mathbf{J}) \tag{4-26}$$

$$\mathbf{R}_x = \mathbf{U}\mathbf{\Lambda}\mathbf{U}^H \tag{4-27}$$

$$\text{MDL}(k) = -\ln\left(\frac{\prod_{j=k+1}^{L}\lambda_j^{\frac{1}{L-k}}}{\frac{1}{L-k}\sum_{j=k+1}^{L}\lambda_j}\right)^{(L-k)M} + \frac{1}{2}k(2L-k)\ln M \tag{4-28}$$

$$\hat{K} = \arg\min_{0 \le k \le L-1} \text{MDL}(k) \tag{4-29}$$

$$K_{final} = \max(1, \min(\hat{K}, K_{max})) \tag{4-30}$$

$$\mathbf{U}_s = [\mathbf{u}_1, \mathbf{u}_2, \cdots, \mathbf{u}_{\hat{K}}] \quad \text{(信号子空间)} \tag{4-31}$$

$$\mathbf{U}_n = [\mathbf{u}_{\hat{K}+1}, \cdots, \mathbf{u}_L] \quad \text{(噪声子空间)} \tag{4-32}$$

$$f_{D,direct} = \min\{f_D^{(k)} : f_D^{(k)} > f_{min} \text{ 且 } f_D^{(k)} < f_{max}\} \tag{4-33}$$

$$\mathbf{X}_1 = \mathbf{X}(1:L-1, :) \tag{4-34}$$

$$\mathbf{X}_2 = \mathbf{X}(2:L, :) \tag{4-35}$$

$$\mathbf{X}_2 = \mathbf{X}_1 \mathbf{\Phi} \tag{4-36}$$

$$\mathbf{U}_{s1} = \mathbf{U}_s(1:L-1, :) \tag{4-37}$$

$$\mathbf{U}_{s2} = \mathbf{U}_s(2:L, :) \tag{4-38}$$

$$\mathbf{\Psi} = (\mathbf{U}_{s1}^H\mathbf{U}_{s1})^{-1}\mathbf{U}_{s1}^H\mathbf{U}_{s2} \tag{4-39}$$

$$\mathbf{\Psi} = \mathbf{V}\mathbf{D}\mathbf{V}^{-1} \tag{4-40}$$

$$f_D^{(k)} = \frac{f_s'}{2\pi}\angle z_k \tag{4-41}$$

$$\mathcal{D} = \{(f_{probe,i}, \tau_{meas,i}, A_i)\}_{i=1}^{N_w} \tag{4-42}$$

## 4.3 基于Metropolis-Hastings的贝叶斯参数反演模型

$$P(\mathbf{\theta} | \mathcal{D}) = \frac{P(\mathcal{D} | \mathbf{\theta}) \cdot P(\mathbf{\theta})}{P(\mathcal{D})} \propto P(\mathcal{D} | \mathbf{\theta}) \cdot P(\mathbf{\theta}) \tag{4-43}$$

$$P(\tau_{meas,i} | \mathbf{\theta}) \propto \exp\left( -\frac{(\tau_{meas,i} - \tau_{model}(f_{probe,i}; \mathbf{\theta}))^2}{2\sigma_0^2} \right) \tag{4-44}$$

$$w_i = \left( \frac{A_i}{\max(\mathbf{A})} \right)^2 \tag{4-45}$$

$$\ln \mathcal{L}(\mathbf{\theta}) = -\frac{1}{2\sigma_0^2} \sum_{i=1}^{N_w} w_i \left( \tau_{meas,i} - \tau_{model}(f_{probe,i}; \mathbf{\theta}) \right)^2 \tag{4-46}$$

$$\tilde{\varepsilon}_r(\omega) = 1 - \frac{\omega_p^2}{\omega(\omega + j\nu_e)} \tag{4-47}$$

$$\tau_{model}(f; n_e, \nu_e) = -\frac{d\Phi}{d\omega} - \frac{d}{c} \tag{4-48}$$

$$P(n_e) = \begin{cases} \frac{1}{n_{e,max} - n_{e,min}} & \text{if } n_e \in [n_{e,min}, n_{e,max}] \\ 0 & \text{otherwise} \end{cases} \tag{4-49}$$

$$P(\nu_e) = \begin{cases} \frac{1}{\nu_{e,max} - \nu_{e,min}} & \text{if } \nu_e \in [\nu_{e,min}, \nu_{e,max}] \\ 0 & \text{otherwise} \end{cases} \tag{4-50}$$

$$\mathbf{\theta}' = \mathbf{\theta}^{(t)} + \boldsymbol{\epsilon}, \quad \boldsymbol{\epsilon} \sim \mathcal{N}(\mathbf{0}, \boldsymbol{\Sigma}_{prop}) \tag{4-51}$$

$$\alpha = \min\left(1, \frac{P(\mathbf{\theta}' | \mathcal{D})}{P(\mathbf{\theta}^{(t)} | \mathcal{D})}\right) = \min\left(1, \exp(\ln\mathcal{L}(\mathbf{\theta}') - \ln\mathcal{L}(\mathbf{\theta}^{(t)}))\right) \tag{4-52}$$

$$\text{CV}_\theta = \frac{\sigma_\theta^{(post)}}{\hat{\theta}^{(post)}} \tag{4-53}$$

## 4.4 Drude等离子体模型仿真验证与不确定性量化

$$\text{SNR}_{\text{dB}} = 10 \log_{10} \frac{P_s}{P_n} \quad \Rightarrow \quad P_n = \frac{P_s}{10^{\text{SNR}_{\text{dB}}/10}} \tag{4-54}$$

$$n(t) \sim \mathcal{N}(0, \sigma_n^2), \quad \sigma_n = \sqrt{P_n} \tag{4-55}$$

$$H_{plasma}(\omega) = \exp\left[-j \cdot \text{Re}\{\tilde{k}(\omega)\} \cdot d - |\text{Im}\{\tilde{k}(\omega)\}| \cdot d\right] \tag{4-56}$$

$$H_{total}(\omega) = e^{-j\omega\tau_{fs}} \cdot H_{plasma}(\omega) \cdot e^{-j\omega\tau_{fs}} = H_{plasma}(\omega)\,e^{-j\omega(2\tau_{fs})} \tag{4-57}$$

$$\delta_k=\frac{A_R-A_L}{A_L+A_C+A_R},\quad f_{beat}=\left(k+\delta_k\right)\frac{f_s}{N} \tag{4-58}$$
